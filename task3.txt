                if ($entry != "." && $entry != "..") {
                    $entryPath = $folderPath . DIRECTORY_SEPARATOR . $entry;

                        // Display folder name

                    } else {
                        // Display file name

                    }
                }
            }

            echo "</ul>";

            // Close the directory handle
